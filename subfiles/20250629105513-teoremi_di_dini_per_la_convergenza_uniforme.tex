\section{Teoremi di Dini per la convergenza uniforme}
\label{sec:org6d8f462}
\begin{thm}
Sia \(f_{n}:[a,b]\to \R\) una \href{20250629105815-successione_di_funzioni.org}{successione di funzioni} \href{20250103103252-funzione_continua.org}{continue}.
\begin{enumerate}
\item Se per ogni \(n \in \N\): \(f_{n+1}\le f_{n}\) e
\begin{equation*}
 \forall x \in [a,b]\ f_{n}(x)\to 0
\end{equation*}
allora \(f_{n}\) \href{20250629105745-convergenza_uniforme.org}{converge uniformemente} a \(0\) su \([a,b]\).
\item Sia \(g:[a,b]\to \R\) continua. Se per ogni \(n \in \N\): \(f_{n+1}\le f_{n}\) e
\begin{equation*}
 \forall x \in [a,b]\ f_{n}(x)\to g(x)
\end{equation*}
allora \(f_{n}\) \href{20250629105745-convergenza_uniforme.org}{converge uniformemente} a \(g\) su \([a,b]\).
\end{enumerate}
\end{thm}
