\section{Teorema di Szeméredi}
\label{sec:orgb0ef069}
\begin{thm}
Per ogni \(A \subseteq \N\) con \uline{\href{20251111151641-densita_di_banach.org}{densità di Banach} positiva} e per ogni \(k<\omega\) esiste una \href{20251111151506-progressione_aritmetica.org}{progressione aritmetica} di lunghezza \(k\) contenuta in \(A\).
\qed
\end{thm}

\(A\) ha \uline{densità di Banach positiva}, se
\begin{equation*}
\limsup_{\substack{%
\card{I}\to \infty\\%
 I\text{ intervallo}}} %
\frac{\card{A\cap I}}{\card{I}} > 0.
\end{equation*}
