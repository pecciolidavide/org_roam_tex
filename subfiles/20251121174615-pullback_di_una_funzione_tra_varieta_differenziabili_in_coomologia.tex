\section{Pullback di una funzione tra varietà differenziabili in coomologia}
\label{sec:org19dd150}
\def\H{H_{\text{dR}}}

Siano \(M,N\) due \href{20250113115909-struttura_differenziabile.org}{varietà differenziabili}, \(F:M\to N\) una \href{20250113144722-funzioni_cinfinito_tra_varieta_differenziabili.org}{funzione \(C^{\infty}\)}.

\begin{definizione}
Poiché il \href{20251115174001-pullback_di_una_funzione_tra_varieta_differenziabili.org}{pullback} commuta con il \href{20251115160537-differenziale_di_una_forma.org}{differenziale}, \(F\) induce un \href{20251201160758-isomorfismo_tra_algebre_graduate.org}{morfismo} tra \href{20251201160758-isomorfismo_tra_algebre_graduate.org}{algebre graduate}:
\begin{align*}
F^{*}: \H^{\bullet}(N) &\longrightarrow \H^{\bullet}(M) \\
[\omega] &\longmapsto F^{*}[\omega] \coloneqq [F^{*}\omega].
\end{align*}
\end{definizione}
