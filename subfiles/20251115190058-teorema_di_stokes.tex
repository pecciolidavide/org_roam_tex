\section{Teorema di Stokes}
\label{sec:org61397a0}
\begin{thm}
Sia \(M\) una \href{20250113115909-struttura_differenziabile.org}{varietà differeniabile} \href{20251223152054-varieta_differenziabile_orientabile.org}{orientabile}, e sia \(\omega\) una \(n-1\)-\href{20251115155511-forma_differenziale_in_un_punto.org}{forma} a \href{20251223153341-supporto_di_una_forma_differenziale.org}{supporto} \href{20250103163701-spazio_topologico_compatto.org}{compatto}. Allora l'\href{20251115185654-integrazione_di_forme_su_varieta_differenziabile_orientata.org}{integrale}:\footnote{\(\dif\) è il \href{20251115160537-differenziale_di_una_forma.org}{differenziale di una forma}.}
\begin{equation*}
\int_{M} \dif \omega = 0.
\end{equation*}
\end{thm}
