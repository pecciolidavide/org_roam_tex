\section{Pettinabilità delle sfere}
\label{sec:orgcc56c61}
\begin{thm}
Sia \(\mathds{S}^{n}\) la \href{20250115150754-sfera_n_dimensionale.org}{sfera \(n\)-dimensionale}.
\(\mathds{S}^{n}\) è \href{20250115145035-varieta_differenziabile_pettinabile.org}{pettinabile} se e solo se \(n\) è dispari.
\end{thm}
\begin{prop}
Se \(n\) è parti, allora non esistono campi vettoriali mai nulli su \(\mathds{S}^{n}\).
\end{prop}
