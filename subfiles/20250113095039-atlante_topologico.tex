\section{Atlante topologico}
\label{sec:org585fd16}
\begin{definizione}
Se \(M\) è una \href{20250111092123-varieta_topologica.org}{varietà topologica}, un \textbf{\textbf{atlante topologico}} di \(M\) è una famiglia
\begin{equation*}
\mathcal{A}\coloneqq\set{(U_{\alpha}, \varphi_{\alpha})}
\end{equation*}
dove \(\set{U_{\alpha}}\) è un \href{20250103164252-ricoprimento.org}{ricoprimento} \href{20250103145124-topologia.org}{aperto} di \(M\) e
\begin{equation*}
\varphi_{\alpha}:U_{\alpha}\longrightarrow V_{\alpha}
\end{equation*}
è una \href{20250111092123-varieta_topologica.org}{carta topologica} per certi \(V_{\alpha} \subseteq \R^{n}\).
\end{definizione}
