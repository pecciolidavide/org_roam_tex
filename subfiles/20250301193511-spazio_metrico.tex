\begin{enumerate}
\item \(d(x,y)\ge 0\);
\item \(d(x,y) = 0\) se e solo se \(x=y\);
\item \(d(x,y) = d(y,x)\);
\item \(d(x,y)\le d(x,z) + d(z,y)\)
\end{enumerate}
