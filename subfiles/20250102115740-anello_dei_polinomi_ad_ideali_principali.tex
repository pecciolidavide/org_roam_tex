\section{Anello dei polinomi ad una indeterminata di un PID è PID}
\label{sec:orgbf90876}
\begin{prop}
Sia \(A\) un \href{20241219112842-pid.org}{PID}. Allora \(A[x]\) \href{20241219113434-anello_dei_polinomi.org}{anello dei polinomi} è un \href{20241219112842-pid.org}{PID}.
\end{prop}
\begin{prop}
Sia \(\K\) un \href{20241205142049-campo.org}{Campo}, e sia \(\K[x]\) l'\href{20241219113434-anello_dei_polinomi.org}{Anello-dei-polinomi} ad una variabile. Allora \(\K[x]\) è un \href{20241219112842-pid.org}{PID}.
\end{prop}
\begin{cor}
Applicando la proposizione 1, per induzione, si ha che \(\K[x_{1},\dots,x_{n}]\) è un \href{20241219112842-pid.org}{PID}.
\end{cor}
