\section{Riduzione della coomologia di un prodotto con la retta reale}
\label{sec:org58905d4}
\begin{thm}
Sia \(M\) una \href{20250113115909-struttura_differenziabile.org}{varietà differenziabile}, e sia
\begin{equation*}
\pi: M\times \R \to M
\end{equation*}
la proiezione. Allora \(\pi\) induce un \href{20251121174615-pullback_di_una_funzione_tra_varieta_differenziabili_in_coomologia.org}{pullback}:
\begin{equation*}
\pi^{*}: H^{\bullet}(M\times \R) \xrightarrow{\cong} H^{\bullet}(M)
\end{equation*}
che è un \href{20251201160758-isomorfismo_tra_algebre_graduate.org}{isomorfismo} tra le \href{20251115175943-prodotto_wedge_in_coomologia_di_de_rham.org}{coomologie}.
\end{thm}
