\subsection{Dimensione dello spazio vettoriale quoziente}
\label{sec:org86d3803}
\begin{prop}
Sia \(V\) \href{20241205142027-spazio_vettoriale.org}{spazio vettoriale} infinito dimensionale, \(W \subseteq V\). Se \(\dim W = n < + \infty\), allora il quoziente \(V/W\) avrÃ  dimensione infinita.
\end{prop}
\begin{proof}
Se per assurdo \(\dim V/W = m\), allora
\begin{equation*}
\frac{V}{W}\oplus W \cong V
\end{equation*}
e, per la \href{20260202230147-formula_di_grassman.org}{Formula di Grassman}, \(\dim V = m+n\). Assurdo.
\end{proof}
