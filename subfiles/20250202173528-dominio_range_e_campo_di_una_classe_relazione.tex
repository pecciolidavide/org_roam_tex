\begin{align*}

\operatorname{ran}(R) &\coloneqq \set{y\,|\,\exists\, x\ (x,y) \in R}\\
\operatorname{fld}(R) &\coloneqq \operatorname{dom}(R)\cup \operatorname{ran}(R)
\end{align*}
