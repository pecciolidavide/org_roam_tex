\section{Modello lambda saturo}
\label{sec:org281580e}
\begin{definizione}
Una \(\mathcal{L}\)-\href{20250131103035-struttura_del_prim_ordine.org}{struttura} \(N\) infinita si dice \uline{\(\lambda\)-saturo} se \href{20250212164424-tipo_teoria_dei_modelli.org}{realizza} ogni tipo in una variabile \(p(x)\) tale che
\begin{enumerate}
\item \(p(x)\) è composto di \href{20250212112324-estensione_di_un_linguaggio_del_prim_ordine.org}{\(\mathcal{L}(A)\)}-\href{20250131103317-formula_del_prim_ordine.org}{formule} per qualche \(A \subseteq N\) di \href{20241213101756-cardinalita.org}{cardinalità} \(<\lambda\);
\item \(p(x)\) è \href{20250212164424-tipo_teoria_dei_modelli.org}{finitamente soddisfacibile in \(N\)}.
\end{enumerate}
\end{definizione}

\begin{prop}
Sono fatti equivalenti:
\begin{enumerate}
\item \(N\) è \(\lambda\)-saturo.
\item \(N\) realizza tutti i \href{20250212164424-tipo_teoria_dei_modelli.org}{tipi} \href{20250212164424-tipo_teoria_dei_modelli.org}{finitamente soddisfacibili} \(p(z)\) di \href{20250212112324-estensione_di_un_linguaggio_del_prim_ordine.org}{\(\mathcal{L}(A)\)}-\href{20250131103317-formula_del_prim_ordine.org}{formule}, con \(\card{z}\le\lambda\) e \(\card{A}<\lambda\)
\end{enumerate}
\end{prop}
\section{Modello saturo}
\label{sec:org9f1a83f}
\(N\) si dice \uline{saturo} se è \(\lambda\)-saturo e \(\card{N}=\lambda\).
