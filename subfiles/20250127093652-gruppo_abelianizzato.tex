\section{Gruppo abelianizzato}
\label{sec:orgb5ba95a}
\begin{definizione}
Sia \(G\) un \href{20241205141146-gruppo_abeliano.org}{gruppo}, e sia \([G,G]\) è il \href{20250127093729-commutatore_di_un_gruppo.org}{sottogruppo commutatore} di \(G\), si definisce il \textbf{gruppo abelianizzato} il seguente \href{20250127093819-quoziente_di_gruppo_e_sottogruppo.org}{quoziente}:
\begin{equation*}
G^{\operatorname{ab}} \coloneqq \frac{G}{[G,G]}
\end{equation*}
\end{definizione}
\begin{prop}
Il gruppo \(G^{\mathrm{ab}}\) è un \href{20250127093245-gruppo_abeliano.org}{gruppo abeliano}.
\end{prop}
