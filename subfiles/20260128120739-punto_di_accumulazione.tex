\section{Punto di accumulazione}
\label{sec:org442323e}
\begin{definizione}
Sia \(X\) uno spazio topologico, \(S \subseteq X\). Un punto \(x_{0} \in X\) è un \uline{punto di accumulazione per \(S\)} se per ogni intorno aperto \(U\) di \(x_{0}\),
\begin{equation*}
(U\setminus\set{x_{0}})\cap S \neq \emptyset.
\end{equation*}
\end{definizione}

\begin{oss}
Non è richiesto che \(x_{0} \in S\).
\end{oss}
