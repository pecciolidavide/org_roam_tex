\section{Morfismo inverso (Teoria delle Categorie)}
\label{sec:org940e909}
\begin{definizione}
Sia \(\mathcal{C}\) una \href{20241126100904-categoria.org}{Categoria}, \(X,Y\) oggetti in \(\mathcal{C}\).
Dato \(f:X\longrightarrow Y\):
\begin{itemize}
\item un \emph{inverso destro} di \(f\) è \(g: Y \longrightarrow X\) tale che
\begin{equation*}
   f\circ g = \1_Y;
\end{equation*}
\item un \emph{inverso sinistro} di \(f\) è \(g: Y \longrightarrow X\) tale che
\begin{equation*}
  g \circ f = \1_X;
\end{equation*}
\item un \emph{inverso bilatero} di \(f\) è \(g: Y \longrightarrow X\) tale che
\begin{equation*}
   f\circ g = \1_Y,\qquad g \circ f = \1_X ;
\end{equation*}
\end{itemize}
\end{definizione}
