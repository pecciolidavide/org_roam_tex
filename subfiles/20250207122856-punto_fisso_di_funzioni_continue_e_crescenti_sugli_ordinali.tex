\section{Punto Fisso di funzioni continue e crescenti sugli ordinali}
\label{sec:org0c0d165}
Contesto: \href{20250130104245-morse_kelly_set_theory.org}{Morse Kelly Set Theory}
\begin{lem}
Sia \(f:\operatorname{Ord}\to \operatorname{Ord}\)\footnote{Vedi ``\href{20250203111003-ordinali.org}{Classe degli ordinali}''} una \href{20250202170607-classe_relazione_binaria.org}{classe-funzione} \href{20250203132953-funzione_monotona.org}{crescente} e \href{20250203161326-topologia_sugli_ordinali.org}{continua}. Allora
\begin{equation*}
\forall\,\alpha\ \exists\,\overline{\alpha}\ (f(\overline{\alpha})=\overline{\alpha}).
\end{equation*}
\end{lem}
