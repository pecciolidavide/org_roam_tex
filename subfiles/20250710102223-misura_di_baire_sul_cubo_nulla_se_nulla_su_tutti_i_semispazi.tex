\section{Misura di Baire sul cubo nulla se nulla su tutti i semispazi}
\label{sec:orgdb63e53}
\begin{prop}
Sia \(\mu\) una \href{20250704163455-misura.org}{misura} \href{20250625104200-misura_di_baire.org}{di Baire} su \(I_{n}\), \href{20250625110024-misura_con_segno.org}{con segno}, \href{20250625110016-misura_finita.org}{finita} e \href{20250625110032-misura_regolare.org}{regolare}.

Se per ogni \(\bm{w} \in \R^{n}\) e per ogni \(\theta \in \R\)
\begin{equation*}
\mu(\mathcal{H}_{\bm{w},\theta}) = 0 = \mu(\mathcal{P}_{\bm{w},\theta})
\end{equation*}
allora \(\mu = 0\), dove
\begin{align*}
\mathcal{H}_{\bm{w},\theta} &= \set{\bm{x} \in \R^{n}\mid \bm{w}\cdot\bm{x}+\theta>0 }
\mathcal{P}_{\bm{w},\theta} &=  \set{\bm{x} \in \R^{n}\mid \bm{w}\cdot\bm{x}+\theta=0 }
\end{align*}
\end{prop}
